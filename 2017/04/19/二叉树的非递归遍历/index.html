  3. 二叉树的非递归遍历

中序遍历非递归：
void InOrder(TreeNode *root){
  if(root == NULL) return ;
  TreeNode *p = root;
  stack<TreeNode> st;

  while(p!=NULL || !st.empty()){
      while(p!=NULL) {
          st.push(p);
          p = p -> left;
      }

      if(!st.empty()){
          p = st.top();
          cout << p -> val << endl;
          st.pop();
          st.push(p->right); 
      }
  }
}
